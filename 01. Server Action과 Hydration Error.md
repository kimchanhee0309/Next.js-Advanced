#### 1. Server Action
`서버환경에서 실행되는 async 함수`

* Form 제출 또는 데이터 변경용 api 요청에 적합
<details>
  <aummary>폼 제출 예시</aummary>

  ```JavaScript
  export default function Page() {
    async function createInvoice(formData) {
      'use server'
      const rawFormData = {
        customerId: formData.get('customerId'),
        amount: formData.get('amount'),
        status: formData.get('status'),
      }

      // mutate data
      // revalidate cache
    }

    return <form action={createInvoice}>...</form>
  }
  ```
</details>

<details>
  <summary>"use server"로 선언된 모듈 내의 async 함수들 예시</summary>

  ```JavaScript
  // action.js
  'use server';  // 파일 최상단에 선언하면 모든 내보내는 함수

  export async function createUser(formData) {
    const name = formData.get('name');
    await db.user.create({ data: { name } });
    return { success: true };
  }

  export async function updateUser(formData) {
    const id = formData.get('id');
    const name = formData.get('name');
    await db.user.update({ where: { id }, data: { name } });
    return { success: true };
  ```
</details>

**서버액션 장점**
* **보안**: api 요청 시 네트워크에 민감한 정보가 노출되지 않고 실행, 브라우저의 보안 위협에서 안전
* **번들 최적화**: 서버액션 코드는 JS 번들파일에 포함되지 않음
* **개발자 경험**: 별도의 api 엔드포인트를 관리할 필요 없음

**서버액션 단점**
* 보안에 민감함 정보를 다루지 않으면서 실시간 유효성 검사가 필요한 경우라면 폼 제출 시 서버액션은 좋은 선택지가 아닐 수 있음
* `클라이언트 → Next.js 서버 → API 서버`로 중간에 Next.js 서버를 거쳐 api 레이턴시가 오히려 더 늘어나 UX 저하의 원인이 될 수 있음

#### 2. Hydration Error
"`서버가 그린 화면`"과 "`클라이언트가 처음 그린 화면`"이 다를 때 나는 에러
→ 서버와 클라이언트 사이에 렌더링 불일치

<details>
  <summary>예시 1: 클라이언트 전용 API 사용</summary>

  ```JavaScript
  'use client'; // 클라이언트 컴포넌트

  function UserLocation() {
    // 서버에서는 window 객체에 접근할 수 없음
    const [location, setLocation] =  useState(window.location.href); // Hydration Error
    const location = window.location.href;

    return <div>현재 URL: {location}</div>;
  }
  ```

  * 해결법: **useEffect를 사용해 클라이언트 측 코드 실행 지연**
    ```JavaScript
    "use client"
    import { useState, useEffect } from "react"

    function UserLocation() {
      const [location, setLocation] = useState("");

      useEffect(() => {
        // 클라이언트에서만 실행
        setTimeout(() => {
          setLocation(window.location.href);
        }, 0);
      }, []);

      return <div>현재 URL: {location}</div>;
    }
    ```
</details>

<details>
  <summary>예시 2: 서버와 클라이언트의 초기 상태 불일치</summary>

  ```javascript
  'use client';

  function RandomNumber() {
    // 서버와 클라이언트에서 각각 다른 값 생성
    const [number] = useState(Math.random()); // Hydration Error

    return <div>랜덤 숫자: {number}</div>;
  }
  ```

  * 해결법: **초기 상태 일치시키기**
    ```javascript
    "use client";

    function RandomNumber() {
      // 서버와 클라이언트에서 각각 다른 값 생성
      const [number, setNumber] = useState(0);
      useEffect(() => {
        setTimeout(() => {
          setNumber(Math.random());
        }, 0);
      }. []);

      return <div>랜덤 숫자: {number}</div>
    }
    ```
</details>

<details>
  <summary>예시 3: 조건부 렌더링 불일치</summary>

  ```javascript
  'use client';

  function ConditionalComponent({ user }) {
    // 서버에는 false, 클라이언트에서는 true가 될 수 있음
    const isLoggedIn = typeof window !== 'undefined' ?
      localStorage.getItem('token') !== null : false; // Hydration Error

    return (
      <div>
        {isLoggedIn ? <p>환영합니다, {user}님!</p> : <p>로그인이 필요합니다</p>}
      </div>
    );
  }
  ```

  * 해결법: **dynamic import로 서버환경에서는 실행되지 않게 조치**
    ```javascript
    "use client";

    export default function ClientComponent({ user }) {
      // 이 컴포넌트는 서버에서 안 그려짐 → localStorage 맘껏 써도 안전
      const isLoggedIn = localStorage.getItem("token") !== null;

      return (
        <div>
          {isLoggedIn ? <p>환영합니다, {user}님!</p> : <p>로그인이 필요합니다</p>
        </div>
      );
    }
    ```
    ```javascript
    "use client";
    import dynamic from "next/dynamic";

    // "서버에선 그리지 말고(ssr:false), 클라에서만 그려라"
    const ClientComponent = dynamic(() => import("./ClientComponent"), {
      ssr: false
    });

    export default function SafeComponent({ user }) {
      return <ClientComponent user={user} />;
    }
    ```
</details>
