#### 프록시(proxy.js)
: 특정 경로에 대한 사용자 요청이 서버에서 처리되기 전에 앞서 어떤 작업을 수행하고 싶을 때 사용하는 함수

#### 알아두면 좋은 NextResponse의 4가지 메소드 의미
```javascript
// 요청을 계속 처리, 단순히 사용자 요청을 그대로 진행시킴
return NextResponse.next();

// about 경로로 redirect 시킴
return NextResponse.redirect(new URL('/about', request.url));

// faq 경로로 rewrite 함(url path는 그대로인데 실행되는 페이지 컴포넌트가 다름)
return NextResponse.rewrite(new URL("/faq", request.url));

// 요청으로 가기 전에 json 데이터를 응답함
return NextResponse.json({ data: { hello: "world" } });
```

#### 권한에 따른 redirect 로직
```javascript
import { NextResponse } from "next/server";

export function proxy(reqeust) {
  if (request.nextUrl.pathname.startsWith("/profile")) {
    const token = request.cookies.get("token")?.value;
    if (!token) {
      // token이 없는 상태로 프로필페이지에 접근 시도 시 홈 화면으로 리다이렉트 시킴
      return NextResponse.redirect(new URL("/", request.url));
    }
    return NextResponse.next();
  }
}

export const config = {
  matcher: [
        "/((?!api|_next/static|_next/image|favicon.ico).*)",
  ],
};
```

#### 권한에 따른 rewrites 로직
```javascript
import { NextResponse } from "next/server";

export funtcion proxy(request) {
  const membershipLevel = request.cookies.get('membershipLeel')?.value || 'guest';
  if (request.nextUrl.pathname.startsWith("/posts")) {
    if (membershipLevel === "admin") {
      // 회원등급이 admin 인 경우, 관리자 전용 페이지를 볼 수 있게 인가처리할 수도 있음
      return NextResponse.rewrite(new URL("/admin/posts", request.url));
    }
    return NextResponse.next();
  }
}

export const config = {
  matcher: [
    "/((?!_next/static|_next/image|favicon.ico).*)",
  ],
};
```

#### 권한 없는 api 요청 시 빠른 응답
```javascrpit
import { NextResponse } from "next/server";

export function proxy(request) {
  if (request.nextUrl.pathname.startsWith("/api/profile")) {
    const token = request.cookies.get("token")?.value;
    if (!token) {
      return NextResponse.json({ message: "token 이 없습니다." });
    }
  }
  return NextResponse.next();
}

export const config = {
  matcher: [
    "/((?!_next/static|_next/image|favicon.ico).*)",
  ],
};

```
